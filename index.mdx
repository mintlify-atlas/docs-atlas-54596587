## Why pkmn/engine?

The engine is **more than 1000× faster** than Pokémon Showdown in compatibility mode — purpose-built for use cases where simulation throughput matters.

<Steps>
  <Step title="Install the package">
    ```bash
    npm install @pkmn/engine
    ```
  </Step>
  <Step title="Create a battle">
    ```ts
    import {Dex} from '@pkmn/dex';
    import {Generations} from '@pkmn/data';
    import {Battle, Choice} from '@pkmn/engine';

    const gens = new Generations(Dex);
    const battle = Battle.create(gens.get(1), {
      seed: [1, 2, 3, 4],
      p1: {name: 'Player 1', team: [/* your team */]},
      p2: {name: 'Player 2', team: [/* your team */]},
      log: false,
    });
    ```
  </Step>
  <Step title="Run the simulation">
    ```ts
    const choose = (choices: Choice[]) =>
      choices[Math.floor(Math.random() * choices.length)];

    let result = battle.update({type: 'pass'}, {type: 'pass'});
    while (!result.type) {
      const c1 = choose(battle.choices('p1', result));
      const c2 = choose(battle.choices('p2', result));
      result = battle.update(c1, c2);
    }
    console.log(result.type); // 'win', 'lose', or 'tie'
    ```
  </Step>
</Steps>
